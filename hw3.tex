\section{Problem 2}

\subsection{Addition}

\((x^5 + x^3 + x^2 + x + 1)\ +\ (x^2 + x + 1)\)

= \(x^5 + x^3\)

\subsection{Subtraction}

\((x^5 + x^3 + x^2 + x + 1)\ -\ (x^2 + x + 1)\)

= \(x^5 + x^3\)

\subsection{Multiplication}

\((x^5 + x^3 + x^2 + x + 1)\ *\ (x^2 + x + 1)\)

\begin{equation*}
  \begin{split}
    x^5 + x^3 + x^2 + x + 1 * x^2 &= x^7 + x^5 + x^4 + x^3 + x^2 \\
    x^5 + x^3 + x^2 + x + 1 * x^2 &= x^6 + x^4 + x^3 + x^2 + x \\
    x^5 + x^3 + x^2 + x + 1 * 1 &= x^5 + x^3 + x^2 + x + 1
  \end{split}
\end{equation*}

\begin{tabular}{ c c c c c c c c }
\(x^7\) &   & \(+\ x^5\) & \(+\ x^4\) & \(+\ x^3\) &\(+\ x^2\)&   &   \\
  & \(+\ x^6\) &   & \(+\ x^4\) & \(+\ x^3\) & \(+\ x^2\) &\(+\ x\) &   \\
  &   & \(+\ x^5\) &   & \(+\ x^3\) & \(+\ x^2\) & \(+\ x\) & \(+\ 1\) \\
\end{tabular}

= \(x^7 + x^6 + x^3 + x^2 + 1\)

\subsection{Division}

\((x^5 + x^3 + x^2 + x + 1)\ /\ (x^2 + x + 1)\)

\polylongdiv{x^5 + x^3 + x^2 + x + 1}{x^2 + x + 1}

= \(x^3 - x^2 + x + 1,\ R\ -x\)
